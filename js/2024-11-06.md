## 树形结构 转化列表


``` javascript
const data = [
    {
        id: 1,
        text: '节点1',
        parentId: 0,
        children: [
            {
                id: 2,
                text: '节点1_1',
                parentId: 1
            }
        ]
    }
]

const treeToList = () => {
  let result = [];
  const dfs = (tree) => {
    tree.forEach((item, index) => {
      if (item.children) {
        // 递归调用
        dfs(item.children);
        delete item.children;
      }
      result.push(item);
    });
  };
  dfs(data)
  return result
};
```

## 判断一个字符串是否为驼峰字符串， judge('ByteDance','BD') -> true judge('Bytedance','BD') -> false


``` javascript 

function judge(str, judge_str) {
  const reg = /[A-Z]/g;

  const matchs = str.match(reg).join("");

  return matchs === judge_str;
}
```

## 压缩字符串  abbccccaaa->a1b2c4a3

``` javascript

// 压缩字符串 abbccccaaa->a1b2c4a3
function pressStr(str) {
  let result = "";
  let cur = str[0];
  let count = 1;

  for (let i = 0; i < str.length; i++) {
    //如果当前字符不等于正确的字符 就更新cur
    if (str[i] !== cur) {
      result += `${cur}${count}`;
      cur = str[i];
      count = 1;
    } else {
      count++;
    }
  }

  result+=`${cur}${count}`;

 

  return result;
}

pressStr("abbccccaaa");


```


##  输入 50a6we8y20x 输出 50 个 a，6 个 we，8 个 y，20 个 x

``` javascript
function recoverStr(str) {
  let res = "";
  let num = 0;
  let cur = "";

  for (let i = 0; i < str.length; i++) {
    if (!isNaN(Number(str[i]))) {
      if (cur) {
        res += cur.repeat(num);
        cur = "";
        num = 0;
      }
      num = num * 10 + Number(str[i]);
    } else {
      cur += str[i];
    }
  }

  res += cur.repeat(num);

  return res;
}

let r = recoverStr("50a6we8y20x");

console.log({

});
```

``` javascript
// Map 场景题

let data = [
  {
    name: "可乐",
    categories: ["热门", "饮料"],
  },
  {
    name: "苹果",
    categories: ["热门", "食物"],

const main = () => {

  let res=[]
  const map = new Map();

  data.forEach((item, index) => {
    const { name, categories } = item;

    categories.forEach((keyWord, index) => {
      if(!map.has(keyWord)){
        map.set(keyWord, [item.name]);
      }else{
        map.get(keyWord).push(item.name);
      }
    });
  });

  map.forEach((val, key) => {
    res.push({ name: key, categories: val });
  });
  return res;
};
```
